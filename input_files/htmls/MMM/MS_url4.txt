	function activeSubTab(){
		/*adds hover action to all tabs not selected*/
		applyTabHover();
		
		/*gets content of preselected tab*/
		getTabContent();
		
		/*checks for a clicked tab*/
		$('ul.in_tabs li a').click(
			function(){
				changeTabs(this);
				getTabContent();
			}
		);
	}

	/*********************************************************************************
	function: 		Adds a hover state to all tabs not selected
	arguments: 		None
	*********************************************************************************/
	function applyTabHover(){
		$('ul.in_tabs li a').not('.tabon').hover(
			function(){
				$(this).addClass('tabhover');
			},
			function(){
				$(this).removeClass('tabhover');
			}
		);
	}
	
	/*********************************************************************************
	function: 		Changes the state of the tabs when you click on them
	arguments: 		The this object (a reference to the link that was clicked
	requirements: 	One tab must currently be selected
	*********************************************************************************/
	function changeTabs(obj){
		/*swaps state of tabs*/
		$(obj).parent().siblings().children('a').removeClass("tabon");
		$(obj).addClass("tabon");
		applyTabHover();
		getTabContent();
	}
	
	/*********************************************************************************
	function: 		Swaps out tab content depending on tab clicked
	arguments: 		None
	requirements: 	One tab must currently be selected
	*********************************************************************************/
	function getTabContent(){
		/*hides all content areas*/
        $('.tab_content').hide();
        /*creates an array and populates it with ids from all tab sections*/
        var tabLists = [];
        $("ul.in_tabs li a.tabon").each(function(){
        	var tabId = $(this).attr('href');
        	//in IE $(this).attr('href') will add absolute path ahead of the tab id, 
        	//if(tabId.indexOf("http") > -1){
        	if(tabId.indexOf("#") > 0) {// if normal tabId.indexOf("#")==0  
        		tabId = tabId.substring(tabId.lastIndexOf("#"));
        	}
            tabLists.push(tabId)
        });
     
        /*loop through each tab section and turn on the correct tab content container*/
        for (var i = 0; i <= tabLists.length - 1; i++) {
            $(tabLists[i]).show();
        }
	}
